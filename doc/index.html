Classy.js

A tiny MVC framework / class system

Provided 'Classy' is a global in your scope, write your Classy modules like this


    function Module(context) {
        var module = Classy(), // this is the object we return at the end
            privateVar; // you can define as many private variables as you like

        function privateFunction() {
            // Has access to all variables, but can't be reached outside this
            // function
        }

        model.extend({
            "init" : function() {
                // It's common to have an 'init' function in your module and
                // let it emit a 'ready' event
                this.emit('ready');
            }
        });

        return module;
    }

Then, use it like this.


    var module = Module({
        "option1" : true
    });

    module.on('ready', function() {
        console.log('woot! ready!');
    });

    module.init();

Using it together with RequireJS and the Javascript server

Classy is a submodule from the util module. You can use it like this:


    // 'module.js'
    require(["vpro/util/${pom.version}/util"], function(util) {
        return function(context) {
            var module = util.Classy();

            module.extend({
                "hello" : function() {
                    return "Hello " + context.name + "!";
                }
            });

            return module;
        };
    });

    // somewhere else
    require(['module'], function(Module) {
        var module = Module({
            "name" : "Elvis"
        });

        console.log( module.hello() ); // 'Hello Elvis'
    });

General methods

extend

module.extend([context, ]object)

Extend your module by giving an object. Keys with the same value will be overwritten. this will be set to the objects context. When setting no context the properties will be inserted in the current object. However, if you want to insert properties 'deeper' into the object, you can specificy that as the first parameter


        module.extend({
            "names" : {
                "woman" : ['claire', 'alice']
            },

            "sayName" : function(i) {
                console.log( "Hello " + this.names.woman[i] + "!" );
            }
        });

        module.extend(module.names, {
            "man" : ['clark', 'elvis']
        });

        console.log(module.names);
        // Gives:
        // {
        //   "woman" : ['claire', 'alice'],
        //   "man" : ['clark', 'elvis']
        // }

Events

on

module.on(eventName, handler, context)

module.on(object, context);

Event listener. Events are only thrown on this object and won't be thrown in other Classy objects. handler will be called with two arguments: data, which may contain any data, and an eventObject that contains some useful information about the event, such as the scope, target and event name


        module.on('ready', function() {
            console.log('your module is ready!');
        });

        module.on({
            "talk" : function() {
                console.log('your module is talking!');
            },

            "walk" : function() {
                console.log('your module is walking!');
            }
        })

emit

module.emit(eventName[, data])

Trigger an event. eventName can be a space seperated string if you want to trigger more events. data can be any Javascript variable you want, and will be passed to any event listeners

        module.extend({
            "sleep" : function() {
                this.emit('sleeping', 'very deep');
            }
        });

Data methods

configureAttributes

module.configureAttributes( attributeArray )

module.configureAttributes( attributeObject )

Add private attributes to the object that can later be modified by using get, set and so on. Note that these won't become a part of your object, you can't reach them simply by using module.keyName;

When given an array all attributes will simply be added. When giving an object you can add a function as the value that will be used for validation

        module.configAttributes(['name']);

        module.set('name', 'Elvis');

        module.get('name'); // 'Elvis'

        module.configAttributes({
            "profession" : function(value) {
                return value === "singer";
            }
        });

        module.set('profession', 'Barber'); // will return 'false'

        module.get('profession'); // 'null'

delete

module.delete( key );

Deletes an attribute. Triggers delete and change events.

get

module.get( key );

Gets an attribute. If the item is not available will return null

getAll

module.getAll();

Returns all attributes as an object.

set

module.set(key, value);

module.set(object);

Sets attributes. You can either set one attribute, or a whole set of attributes by using an object.

Every attribute will trigger a change event. A key that doesn't exist will trigger a create event, a key that does exist will trigger an update event.

All events will have the key of the attribute as their event value.


            module.on({
                "change" : function(key) {
                    console.log('Something happened with ' + key);
                },

                "create" : function(key) {
                    console.log("New attribute " + key + " added!");
                },

                "update" : function(key) {
                    console.log("Attribute " + key + " was updated!");
                }
            });

            module.set('name', 'Elvis'); // will trigger 'change' and 'create' events
            module.set('name', 'Johnny'); // will trigger 'change' and 'update' events

filter

module.filter( function );

Gets an attribute using a custom function.


            module.set('singer', {
                'name' : 'Elvis',
                'instrument' : 'Guitar'
            });

            var singer = module.filter(function(item) {
                return item.name === "Elvis";
            });

has

module.has( key );

Checks if a key is available and returns true or false.
